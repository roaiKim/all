```ts
import { useMemo } from "react";

class fullScreenModal {
    currentFullScreen: any;
    constructor() {
        this.currentFullScreen = null;
    }

    requestFullscreen(element) {
        if (!element) {
            return;
        }
        // 进入全屏
        if (element.requestFullscreen) {
            element.requestFullscreen();
        } else if (element.webkitRequestFullscreen) {
            element.webkitRequestFullscreen();
        } else if (element.mozRequestFullScreen) {
            element.mozRequestFullScreen();
        } else if (element.msRequestFullscreen) {
            element.msRequestFullscreen();
        }
    }

    exitFullscreen(element) {
        if (!element) {
            return;
        }
        if (element.exitFullscreen) {
            element.exitFullscreen();
        } else if (element.webkitExitFullscreen) {
            element.webkitExitFullscreen();
        } else if (element.mozCancelFullScreen) {
            element.mozCancelFullScreen();
        } else if (element.msExitFullscreen) {
            element.msExitFullscreen();
        }
    }

    hasCurrentRequestFullscreen() {
        return !!this.currentFullScreen?.fullscreenElement;
    }

    isFullscreen(element) {
        if (!element) {
            return false;
        }
        return !!element.fullscreenElement;
    }

    toggleFullscreen(element) {
        if (!element) {
            return;
        }
        // 如果当前有全屏 先取消
        if (this.hasCurrentRequestFullscreen()) {
            this.exitFullscreen(this.currentFullScreen);
        }
        if (this.isFullscreen(element)) {
            this.currentFullScreen = element;
            this.exitFullscreen(element);
        } else {
            this.currentFullScreen = null;
            this.requestFullscreen(element);
        }
    }
}

export function useFullScreen() {
    const fullScreen = useMemo(() => {
        return new fullScreenModal();
    }, []);

    return { fullScreen };
}
```
